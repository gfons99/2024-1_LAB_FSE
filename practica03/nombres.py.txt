from bluedot import BlueDot
from signal import pause
from RPLCD import CharLCD

lcd = CharLCD(cols=16, rows=2, pin_rs=26, pin_e=19 pins_data=[13,6,5,11])

def dpad=(pos):
	if pos.top:
	  	lcd.write_string(u'Fernanda')
		sleep(2)
	elif pos.bottom:
	  	lcd.write_string(u'Gadiel')
		sleep(2)
	elif pos.left:
	  	lcd.write_string(u'Bryan')
		sleep(2)

bd = BlueDot()
bd.when_pressed = dpad

pause()
