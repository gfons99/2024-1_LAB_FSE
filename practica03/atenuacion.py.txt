from bluedot import BlueDot
from signal import pause
from gpiozero import PWMLED


red = PWMLED(17)
green = PWMLED(27)
yellow = PWMLED(22)

def set_brightness(pos):

	if pos.left:
		red.value = 0
		green.value = 0.5
		yellow.value = 1
	elif pos.rigth:
		red.value = 1
		green.value = 0.5
		yellow.value = 0
	elif pos.bottom:
		red.value = 0.5
		green.value = 1
		yellow.value = 0.5

bd = BlueDot()
bd.when_moved = set_brightness

pause()
